document.addEventListener('DOMContentLoaded', () => {
    // Configuration
    const BLOG_ID = "2705228982741823649";
    const POSTS_PER_PAGE = 12;

    const state = {
        allJobs: [],
        filteredJobs: [],
        currentPage: 1,
        categories: new Set(),
        modalTimer: null,
        nextStartIndex: 1,
        isFetchingFromServer: false,
        hasMoreServerJobs: true
    };

    // DOM Elements Cache
    const DOM = {
        statsNumber: document.getElementById('statsNumber'),
        searchInput: document.getElementById('searchInput'),
        categoryFilter: document.getElementById('categoryFilter'),
        dateFilter: document.getElementById('dateFilter'),
        sortBy: document.getElementById('sortBy'),
        jobsContainer: document.getElementById('jobsContainer'),
        paginationContainer: document.getElementById('paginationContainer'),
        jobModal: document.getElementById('jobModal'),
        modalClose: document.querySelector('.modal-close'),
        
        // Modal internals
        modalTitle: document.getElementById('modalTitle'),
        modalMeta: document.getElementById('modalMeta'),
        modalTags: document.getElementById('modalTags'),
        modalBody: document.getElementById('modalBody'),
        modalApplyBtn: document.getElementById('modalApplyBtn')
    };

    // Compiled Regex Patterns
    const urlPatterns = [
        /href="(https:\/\/www\.linkedin\.com\/jobs\/view\/[^"]+)"/i,
        /href="(https:\/\/forms\.gle\/[^"]+)"/i,
        /href="(mailto:[^"]+)"/i,
        /href="(https:\/\/[^"]+apply[^"]*)"/i,
        /href="(https:\/\/[^"]+career[^"]*)"/i,
        /href="(https:\/\/[^"]+job[^"]*)"/i
    ];
    const externalLinkFallbackPattern = /href="(https?:\/\/(?!www\.blogger\.com|blogspot\.com)[^"]+)"/i;
    const htmlTagsRemover = /<[^>]*>/g;
    const spacesShrinker = /\s+/g;

    function extractApplyLink(content) {
        const temp = document.createElement('div');
        temp.innerHTML = content;
        let applyUrl = null;

        // 1. Scrape specific anchor tags containing the word 'apply' securely from their inner content
        const anchors = Array.from(temp.querySelectorAll('a'));
        for (const a of anchors) {
            const html = (a.innerHTML || '').toLowerCase();
            const href = a.getAttribute('href');
            if (href && html.includes('apply') && !href.includes('whatsapp') && !html.includes('whatsapp')) {
                applyUrl = href;
                break;
            }
        }

        // 2. Discover stealthy image buttons functioning as links
        if (!applyUrl) {
            for (const a of anchors) {
                const href = a.getAttribute('href');
                if (!href || href.includes('whatsapp')) continue;
                
                const imgs = a.querySelectorAll('img');
                for (const img of imgs) {
                    const txt = ((img.getAttribute('src') || '') + " " + (img.getAttribute('alt') || '')).toLowerCase();
                    if (txt.includes('apply') || txt.includes('btn') || txt.includes('button')) {
                        applyUrl = href;
                        break;
                    }
                }
                if (applyUrl) break;
            }
        }

        // 3. Fallback to Regex searching for raw unformatted text URLs containing career keywords
        if (!applyUrl) {
            for (let pattern of urlPatterns) {
                const match = content.match(pattern);
                if (match && !match[1].includes('blogger') && !match[1].includes('blogspot') && !match[1].includes('whatsapp')) {
                    applyUrl = match[1];
                    break;
                }
            }
        }
        
        // 4. Ultimate fallback to the first raw external URL found (avoids internal loops)
        if (!applyUrl) {
           const externalMatch = content.match(externalLinkFallbackPattern);
           if (externalMatch && !externalMatch[1].includes('whatsapp')) {
               applyUrl = externalMatch[1];
           }
        }

        return applyUrl || '#';
    }

    function extractExcerpt(content, maxLength = 150) {
        const text = content.replace(htmlTagsRemover, ' ').replace(spacesShrinker, ' ').trim();
        return text.length > maxLength ? text.substring(0, maxLength) + '...' : text;
    }

    const NOW = Date.now();
    const MS_PER_DAY = 1000 * 60 * 60 * 24;
    function formatDate(date) {
        const diff = NOW - date.getTime();
        const days = Math.floor(diff / MS_PER_DAY);

        if (days === 0) return 'Today';
        if (days === 1) return 'Yesterday';
        if (days < 7) return `${days} days ago`;
        return date.toLocaleDateString();
    }

    function fetchJobs(startIndex = 1, maxResults = 40) {
        return new Promise((resolve, reject) => {
            const callbackName = 'bloggerCallback_' + Date.now() + '_' + Math.random().toString(36).substring(2, 9);
            
            const timeoutId = setTimeout(() => {
                delete window[callbackName];
                if (script && script.parentNode) script.parentNode.removeChild(script);
                reject(new Error('Connection to the database timed out. Please check your network or ad-blocker.'));
            }, 8000);

            window[callbackName] = (data) => {
                clearTimeout(timeoutId);
                delete window[callbackName];
                if (script.parentNode) script.parentNode.removeChild(script);
                resolve(data);
            };

            const script = document.createElement('script');
            script.src = `https://www.blogger.com/feeds/${BLOG_ID}/posts/default?alt=json-in-script&callback=${callbackName}&start-index=${startIndex}&max-results=${maxResults}`;
            script.onerror = () => {
                clearTimeout(timeoutId);
                delete window[callbackName];
                if (script.parentNode) script.parentNode.removeChild(script);
                reject(new Error('Failed to load jobs from Server'));
            };
            document.head.appendChild(script);
        });
    }

    async function fetchServerJobs(chunks = 5) {
        if (state.isFetchingFromServer || !state.hasMoreServerJobs) return [];
        state.isFetchingFromServer = true;
        
        const entries = [];
        try {
            for (let i = 0; i < chunks; i++) {
                try {
                    const data = await fetchJobs(state.nextStartIndex, 40);
                    const chunk = data.feed?.entry || [];
                    entries.push(...chunk);
                    state.nextStartIndex += 40;
                    if (chunk.length === 0) {
                        state.hasMoreServerJobs = false; 
                        break;
                    }
                } catch (err) {
                    console.warn('Chunk fetch failed:', err);
                    break;
                }
            }
        } finally {
            state.isFetchingFromServer = false;
        }
        return entries;
    }

    function processEntries(entries) {
        const newJobs = entries.map(entry => {
            const content = entry.content?.$t || '';
            const categoriesList = entry.category?.map(c => c.term) || [];
            const pubDateStr = entry.published?.$t;
            const dateObj = new Date(pubDateStr);
            const title = entry.title?.$t || 'Untitled Job';
            const excerpt = extractExcerpt(content);
            
            categoriesList.forEach(cat => state.categories.add(cat));

            const techKeywords = ['remote', 'senior', 'lead', 'engineer', 'developer', 'manager', 'full stack', 'frontend', 'backend', 'hiring'];
            let score = 0;
            const textForScoring = (title + ' ' + excerpt).toLowerCase();
            techKeywords.forEach(kw => {
                if (textForScoring.includes(kw)) score += 1;
            });

            const daysOld = Math.floor((NOW - dateObj.getTime()) / MS_PER_DAY);
            if (daysOld <= 2) score += 2;
            else if (daysOld <= 7) score += 1;

            return {
                id: entry.id?.$t || '',
                title: title,
                content: content,
                excerpt: excerpt,
                applyLink: extractApplyLink(content),
                categories: categoriesList,
                author: entry.author?.[0]?.name?.$t || 'Unknown',
                date: dateObj,
                dateFormatted: formatDate(dateObj),
                isoDate: pubDateStr,
                score: score,
                daysOld: daysOld
            };
        });

        // Add dynamically fetched items into main memory and instantly render them
        state.allJobs.push(...newJobs);
        updateCategoryFilter();
        applyFilters();
    }

    async function loadJobs() {
        try {
            const entries = await fetchServerJobs(1); // Instant load of the first ~40 items using only 1 sequential fetch to maximize speed

            if (entries.length === 0 && state.allJobs.length === 0) {
                DOM.jobsContainer.innerHTML = `
                    <div class="no-results">
                        <i class="fas fa-database"></i>
                        <h3>No jobs found</h3>
                        <p>The feed returned no data currently.</p>
                    </div>
                `;
                return;
            }

            processEntries(entries);

        } catch (error) {
            console.error('Error fetching jobs:', error);
            DOM.jobsContainer.innerHTML = `
                <div class="no-results">
                    <i class="fas fa-exclamation-triangle"></i>
                    <h3>Unable to pull data</h3>
                    <p>${escapeHtml(error.message)}</p>
                    <button onclick="location.reload()" style="margin-top: 15px; padding: 10px 20px; background: var(--primary-color); color: white; border: none; border-radius: 8px; cursor: pointer;">
                        Retry Now
                    </button>
                </div>
            `;
        }
    }

    async function loadMoreFromServer(btnElement) {
        if (state.isFetchingFromServer) return;
        
        const originalText = btnElement.innerHTML;
        btnElement.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Fetching records securely...';
        btnElement.style.pointerEvents = 'none';
        
        const entries = await fetchServerJobs(3); // Grab another block of 120 historic records per click
        
        if (entries.length > 0) {
            processEntries(entries);
        } else {
            // Restore button locally if fetch failed but wasn't empty
            btnElement.innerHTML = originalText;
            btnElement.style.pointerEvents = 'auto';
            renderJobs(); 
        }
    }

    function updateCategoryFilter() {
        const sortedCategories = Array.from(state.categories).sort();
        const currentSelection = DOM.categoryFilter.value;
        
        DOM.categoryFilter.innerHTML = '<option value="">All Categories</option>' +
            sortedCategories.map(cat => `<option value="${escapeHtml(cat)}" ${currentSelection === cat ? 'selected' : ''}>${escapeHtml(cat)}</option>`).join('');
    }

    function updateStats() {
        DOM.statsNumber.textContent = `${state.filteredJobs.length} Jobs Available`;
    }

    function applyFilters() {
        const searchTerm = DOM.searchInput.value.toLowerCase();
        const category = DOM.categoryFilter.value;
        const dateFilter = DOM.dateFilter.value;
        const sortBy = DOM.sortBy.value;

        state.filteredJobs = state.allJobs.filter(job => {
            const matchesSearch = !searchTerm ||
                job.title.toLowerCase().includes(searchTerm) ||
                job.excerpt.toLowerCase().includes(searchTerm) ||
                job.categories.some(cat => cat.toLowerCase().includes(searchTerm));

            const matchesCategory = !category || job.categories.includes(category);

            let matchesDate = true;
            if (dateFilter === '7') {
                matchesDate = job.daysOld <= 7;
            } else if (dateFilter === '30') {
                matchesDate = job.daysOld <= 30;
            }

            return matchesSearch && matchesCategory && matchesDate;
        });

        if (sortBy === 'title') {
            state.filteredJobs.sort((a, b) => a.title.localeCompare(b.title));
        } else if (sortBy === 'best') {
            state.filteredJobs.sort((a, b) => {
                if (b.score !== a.score) return b.score - a.score;
                return b.date - a.date; 
            });
        } else {
            state.filteredJobs.sort((a, b) => b.date - a.date);
        }

        updateStats();
        renderJobs();
    }

    function renderJobs() {
        const totalPages = Math.ceil(state.filteredJobs.length / POSTS_PER_PAGE);
        
        // Safety bound check in case list physically shrinks after a filter change
        if (state.currentPage > totalPages && totalPages > 0) state.currentPage = totalPages;
        
        const start = (state.currentPage - 1) * POSTS_PER_PAGE;
        const pageJobs = state.filteredJobs.slice(start, start + POSTS_PER_PAGE);

        if (pageJobs.length === 0) {
            let msg = "Try adjusting your search criteria or date filters.";
            if (state.hasMoreServerJobs) {
                msg = `<button class="apply-btn js-load-server-btn" style="background: var(--text-color); margin-top: 15px;">
                        <i class="fas fa-search"></i> Search Deeper (Load Older Database Entries)
                       </button>`;
            }

            DOM.jobsContainer.innerHTML = `
                <div class="no-results">
                    <i class="fas fa-search"></i>
                    <h3>No matching jobs in local memory</h3>
                    <p>${msg}</p>
                </div>
            `;
            DOM.paginationContainer.innerHTML = '';
            return;
        }

        DOM.jobsContainer.innerHTML = `
            <div class="jobs-grid" id="jobsGrid">
                ${pageJobs.map(job => `
                    <div class="job-card" data-job-id="${escapeHtml(job.id)}">
                        <h3 class="job-title">${escapeHtml(job.title)}</h3>
                        <div class="job-meta">
                            <span><i class="fas fa-building"></i> ${escapeHtml(job.author)}</span>
                            <span><i class="fas fa-calendar-alt"></i> ${job.dateFormatted}</span>
                        </div>
                        <div class="job-tags">
                            ${job.categories.slice(0, 3).map(cat => `<span class="tag">${escapeHtml(cat)}</span>`).join('')}
                            ${job.categories.length > 3 ? `<span class="tag">+${job.categories.length - 3}</span>` : ''}
                        </div>
                        <div class="job-excerpt">
                            ${escapeHtml(job.excerpt)}
                        </div>
                        <div class="job-footer">
                            <span class="apply-btn js-apply-btn">View Job <i class="fas fa-arrow-right"></i></span>
                            <span class="date"><i class="fas fa-clock"></i> ${job.dateFormatted}</span>
                        </div>
                    </div>
                `).join('')}
            </div>
        `;

        renderPagination(totalPages);
    }

    DOM.jobsContainer.addEventListener('click', (e) => {
        const card = e.target.closest('.job-card');
        if (card) {
            const jobId = card.dataset.jobId;
            const job = state.filteredJobs.find(j => j.id === jobId);
            if (job) showJobModal(job);
        }
        
        // If clicking global search deeper button in empty state
        const loadBtn = e.target.closest('.js-load-server-btn');
        if (loadBtn) {
            loadMoreFromServer(loadBtn);
        }
    });

    DOM.paginationContainer.addEventListener('click', (e) => {
        const btn = e.target.closest('.page-btn');
        if (btn && btn.dataset.page) {
            state.currentPage = parseInt(btn.dataset.page, 10);
            renderJobs();
            window.scrollTo({ top: document.querySelector('.search-section').offsetTop - 20, behavior: 'smooth' });
        }
        
        const loadBtn = e.target.closest('.js-load-server-btn');
        if (loadBtn) {
            loadMoreFromServer(loadBtn);
        }
    });

    function showJobModal(job) {
        if (state.modalTimer) {
            clearInterval(state.modalTimer);
            state.modalTimer = null;
        }

        // Secure Content Scrubbing
        let optimizedContent = job.content;

        // Basic string-level safety scrubs for bloated embeds
        optimizedContent = optimizedContent
            .replace(/<img /ig, '<img loading="lazy" ')
            .replace(/(you have to wait \d+ seconds|generating apply link(\.{3,})?)/ig, '');

        // Advanced, non-destructive DOM manipulation to strip duplicate buttons and un-hide real content
        let tempDiv = document.createElement('div');
        tempDiv.innerHTML = optimizedContent;

        // Force ANY hidden job descriptions sent by the Blogger payload to become fully visible immediately
        tempDiv.querySelectorAll('[style*="display: none"], [style*="display:none"]').forEach(el => {
            el.style.display = 'block';
            el.style.opacity = '1';
        });

        // Safely scrub duplicate "Apply" elements but HIJACK the Whatsapp tracking to your own channel
        tempDiv.querySelectorAll('a, button').forEach(el => {
            const htmlLower = el.innerHTML.toLowerCase();
            const rawHref = el.getAttribute('href') || '';
            const resolvedHref = el.href || '';
            
            if (
                htmlLower.includes('apply') || 
                (job.applyLink && job.applyLink !== '#' && (rawHref.trim() === job.applyLink.trim() || resolvedHref.trim() === job.applyLink.trim()))
            ) {
                el.remove();
            } else if (rawHref.includes('whatsapp') || htmlLower.includes('whatsapp')) {
                // Divert to own link
                el.href = 'https://whatsapp.com/channel/0029VbBqVvsDzgTKQP5NtJ0t/100';
                el.target = '_blank';
                el.rel = 'noopener noreferrer';
            }
        });

        // Aggressively scrub any buttons disguised as pure images (but spare the hijacked WhatsApp buttons)
        tempDiv.querySelectorAll('img').forEach(img => {
            const txt = (img.src + " " + img.alt).toLowerCase();
            if (txt.includes('whatsapp')) return; // Spare whatsapp tracking icons
            if (txt.includes('apply') || txt.includes('btn') || txt.includes('button')) {
                img.remove();
            }
        });

        // Destroy leftover garbage texts that got un-hidden but were split across multiple HTML nodes
        const textWalker = document.createTreeWalker(tempDiv, NodeFilter.SHOW_TEXT);
        let n;
        while ((n = textWalker.nextNode())) {
            const val = n.nodeValue.toLowerCase();
            if (val.includes('generating') && val.includes('link')) n.nodeValue = '';
            if (val.includes('wait') && val.includes('second')) n.nodeValue = '';
        }

        optimizedContent = tempDiv.innerHTML;

        DOM.modalTitle.innerHTML = escapeHtml(job.title);
        DOM.modalMeta.innerHTML = `
            <span><i class="fas fa-building"></i> ${escapeHtml(job.author)}</span>
            <span><i class="fas fa-calendar-alt"></i> ${new Date(job.isoDate).toLocaleString()}</span>
        `;
        DOM.modalTags.innerHTML = job.categories.map(cat => `<span class="tag">${escapeHtml(cat)}</span>`).join('');
        DOM.modalBody.innerHTML = optimizedContent;

        // Reset Button State
        DOM.modalApplyBtn.className = 'timer-box';
        DOM.modalApplyBtn.innerHTML = `<i class="fas fa-spinner fa-spin"></i> Preparing your link...`;
        DOM.modalApplyBtn.href = "javascript:void(0)";
        DOM.modalApplyBtn.target = "_self";
        DOM.modalApplyBtn.rel = "";

        let timeLeft = 20;

        state.modalTimer = setInterval(() => {
            timeLeft--;
            if (timeLeft > 0) {
                DOM.modalApplyBtn.innerHTML = `<i class="fas fa-hourglass-half"></i> Security check: ${timeLeft}s remaining...`;
            } else {
                clearInterval(state.modalTimer);
                state.modalTimer = null;
                DOM.modalApplyBtn.className = 'timer-box active';
                DOM.modalApplyBtn.innerHTML = `Apply Responsibly Now <i class="fas fa-external-link-alt"></i>`;
                DOM.modalApplyBtn.href = job.applyLink;
                DOM.modalApplyBtn.target = "_blank";
                DOM.modalApplyBtn.rel = "noopener noreferrer";
            }
        }, 1000);

        DOM.jobModal.classList.add('active');
    }

    function renderPagination(totalPages) {
        let html = '';

        if (totalPages > 1) {
            if (state.currentPage > 1) {
                html += `<button class="page-btn" data-page="${state.currentPage - 1}"><i class="fas fa-chevron-left"></i></button>`;
            }

            const maxVisible = 5;
            let startPage = Math.max(1, state.currentPage - Math.floor(maxVisible / 2));
            let endPage = Math.min(totalPages, startPage + maxVisible - 1);

            if (endPage - startPage + 1 < maxVisible) {
                startPage = Math.max(1, endPage - maxVisible + 1);
            }

            if (startPage > 1) {
                html += `<button class="page-btn" data-page="1">1</button>`;
                if (startPage > 2) html += `<span class="page-btn" style="pointer-events:none; border:none; background:none;">...</span>`;
            }

            for (let i = startPage; i <= endPage; i++) {
                html += `<button class="page-btn ${i === state.currentPage ? 'active' : ''}" data-page="${i}">${i}</button>`;
            }

            if (endPage < totalPages) {
                if (endPage < totalPages - 1) html += `<span class="page-btn" style="pointer-events:none; border:none; background:none;">...</span>`;
                html += `<button class="page-btn" data-page="${totalPages}">${totalPages}</button>`;
            }

            if (state.currentPage < totalPages) {
                html += `<button class="page-btn" data-page="${state.currentPage + 1}"><i class="fas fa-chevron-right"></i></button>`;
            }
        }
        
        // Dynamic loading element integration
        let serverLoadHtml = '';
        if (state.hasMoreServerJobs) {
            serverLoadHtml = `
                <div style="flex-basis: 100%; text-align: center; margin-top: 2rem;">
                    <button class="apply-btn js-load-server-btn" style="background: var(--text-color);">
                        <i class="fas fa-cloud-download-alt"></i> Load Older Jobs From Database
                    </button>
                    ${totalPages > 0 ? '<p style="margin-top: 10px; font-size: 0.85rem; color: #64748b;">Expand the archive for more results.</p>' : ''}
                </div>
            `;
        }

        DOM.paginationContainer.innerHTML = html + serverLoadHtml;
    }

    function escapeHtml(text) {
        if (!text) return '';
        const map = {
            '&': '&amp;',
            '<': '&lt;',
            '>': '&gt;',
            '"': '&quot;',
            "'": '&#039;'
        };
        return text.toString().replace(/[&<>"']/g, function(m) { return map[m]; });
    }

    // Handles distinct reset-to-page-1 behavior strictly on structural user query changes
    function applyFiltersReset() {
        state.currentPage = 1; 
        applyFilters();
    }

    function setupEventListeners() {
        DOM.searchInput.addEventListener('input', applyFiltersReset);
        DOM.categoryFilter.addEventListener('change', applyFiltersReset);
        DOM.dateFilter.addEventListener('change', applyFiltersReset);
        DOM.sortBy.addEventListener('change', applyFiltersReset);

        const closeModal = () => {
            DOM.jobModal.classList.remove('active');
            if (state.modalTimer) {
                clearInterval(state.modalTimer);
                state.modalTimer = null;
            }
        };

        DOM.modalClose.addEventListener('click', closeModal);

        DOM.jobModal.addEventListener('click', (e) => {
            if (e.target === DOM.jobModal) closeModal();
        });

        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && DOM.jobModal.classList.contains('active')) closeModal();
        });
    }

    function init() {
        setupEventListeners();
        loadJobs();
    }

    init();
});
